a < b < c の条件を維持しながら a + b + c = 1000 なるピタゴラス数を 順番に探していく

def sum_of_pythagoras(sum)
  a = 1; b, c = a+1, a+2
  limit = sum
  loop do
    if pythagoras?(a, b, c) and (a + b + c) == sum
      return a, b, c
    end
    c += 1
    if c > limit
      b += 1; c = b + 1
    end
    if b > limit/2
      a += 1; b = a + 1; c = b + 1
    end
    return nil if a > limit/3
  end
end
def pythagoras?(a, b, c)
  return true if (a ** 2 + b ** 2) == c ** 2
  false
end

 def greatest_five(number)
   seq = number.split("").map { |s| s.to_i  }
   candidate = 0
  (seq.length-4).times do |n|
     multi = seq.slice(n, 5).inject(:*)
     candidate = multi if multi > candidate
   end
   candidate
 end

(comment)

a + b + c = n
を
c = n - (a + b)
と変形して、
a² + b² = c²
に代入して、さらに変形すると
b = (2 * a * n - n²) / (2 * a - 2 *n)
= n - n² / (2 * (n - a))
となります。
この式の a に 1 から順に整数を代入していって、b が整数になる場合を求め
ると、かなり速く答えが出せます。


def pythagorean_triples (n)
return([]) if n.odd?

ans = Array.new
1.step(n) do |a|
b = n - 0.5 * (n 2) / (n - a.to_f)
case
when (a > b)
return(ans)
when (b.to_i == b)
ans.push([a, b.to_i, Math.sqrt(a 2 + b ** 2).to_i])
end
end
end
